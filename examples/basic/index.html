controller

$('.controll-box').klaster({
    delay: 0,
    actions: {
        'filterbutton': {
            'click': function(e) {
                var $checks = $($(this).attr('data-connected'));
                $checks.toggleOmit().toggle(200);
                return $checks.attr('data-omit');
            }
        }
    },
    'sync': function(el) {
        $('#json-preview').html(JSON.stringify(this.values));
        //prettyPrint();
    }
});
